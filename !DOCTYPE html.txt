<script type="module">
    import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js";
    import { getDatabase, ref, push, onChildAdded, serverTimestamp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-database.js";

    // あなたのFirebaseの設定（スクリーンショットの内容を反映しました）
    const firebaseConfig = {
        apiKey: "AIzaSyC0LabfY8QcZgSFySBrF5Jj6xnKMbPXbZo",
        authDomain: "my-chat-1d626.firebaseapp.com",
        projectId: "my-chat-1d626",
        storageBucket: "my-chat-1d626.firebasestorage.app",
        messagingSenderId: "769979957640",
        appId: "1:769979957640:web:777fb6120b5b223e127a3d"
    };

    const app = initializeApp(firebaseConfig);
    const db = getDatabase(app);
    const chatRef = ref(db, "chat");

    let myName = "";

    // 名前を入力して入室
    window.enterChat = function() {
        const nameInput = document.getElementById('user-id');
        if (nameInput.value.trim() !== "") {
            myName = nameInput.value;
            document.getElementById('login-screen').style.display = 'none';
            document.getElementById('chat-container').style.display = 'flex';
        }
    };

    // 送信ボタンを押した時
    document.getElementById('send-btn').addEventListener('click', () => {
        const input = document.getElementById('input-message');
        const text = input.value.trim();
        if (text !== "") {
            push(chatRef, {
                name: myName,
                text: text,
                time: serverTimestamp()
            });
            input.value = "";
        }
    });

    // 新しいメッセージが届いたら画面に表示
    onChildAdded(chatRef, (data) => {
        const msg = data.val();
        const chatScreen = document.getElementById('chat-screen');
        const side = msg.name === myName ? "right" : "left";
        
        const msgDiv = document.createElement('div');
        msgDiv.classList.add('message-box', side);
        msgDiv.innerHTML = `
            <div class="user-name">${msg.name}:</div>
            <div class="balloon">${msg.text}</div>
        `;
        chatScreen.appendChild(msgDiv);
        chatScreen.scrollTop = chatScreen.scrollHeight;
    });
</script>
