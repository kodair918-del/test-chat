    <div class="login-box">
        <h2>チャットを始める</h2>
        <input type="text" id="user-id" placeholder="あなたの名前を入力">
        <button onclick="enterChat()">入室する</button>
    </div>

<div id="chat-container">
    <div id="chat-screen">
        <div class="message-box left">
            <div class="user-name">Asann:</div>
            <div class="balloon">〇〇おいしいよね～</div>
        </div>
    </div>
    <div class="input-area">
        <input type="text" id="input-message" placeholder="メッセージを入力...">
        <button onclick="sendMessage()">送信</button>
    </div>
</div>

<script>
    let myName = "";
    function enterChat() {
        const nameInput = document.getElementById('user-id');
        if (nameInput.value.trim() !== "") {
            myName = nameInput.value;
            document.getElementById('login-screen').style.display = 'none';
            document.getElementById('chat-container').style.display = 'flex';
        }
    }
    function sendMessage() {
        const input = document.getElementById('input-message');
        const text = input.value.trim();
        if (text !== "") {
            const chatScreen = document.getElementById('chat-screen');
            const msgDiv = document.createElement('div');
            msgDiv.classList.add('message-box', 'right');
            msgDiv.innerHTML = `<div class="balloon">${text}</div>`;
            chatScreen.appendChild(msgDiv);
            input.value = "";
            chatScreen.scrollTop = chatScreen.scrollHeight;
        }
    }
</script>
